CREATE TABLE char_table(
    campaign_id text,
    char_id text,
    char_name text,
    char_class text,
    level smallint,
    align text, 
    max_hp smallint, 
    curr_hp smallint,
    CHR smallint,

    INT smallint,
    WIS smallint,
    DEX smallint,
    CON smallint,
    AC smallint,
    /*prof_bonus smallint,

    //merging from char_map
    spells list<text>,
